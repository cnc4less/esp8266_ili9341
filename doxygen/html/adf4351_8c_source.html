65 uint32_t ADF4351_GetReg32(int addr)
66 {
67  uint32_t R;
68  switch (addr)
69  {
70  case 0 :
71  R = 0
72  | ((regs.r0.FRAC & 0x0fff ) << 3 )
73  | ((regs.r0.INT & 0xffff ) << 15 );
74  return(R );
75  case 1 :
76  R = 1
77  | (( regs.r1.MOD & 0x0fff ) << 3 )
78  | (( regs.r1.Phase & 0x0fff ) << 15 )
79  | (( regs.r1.Prescaler & 1 ) << 27 )
80  | (( regs.r1.PhaseAdjust & 1 ) << 28 );
81  return(R );
82  case 2 :
83  R = 2
84  | (( regs.r2.Reset_R_N & 1 ) << 3 )
85  | (( regs.r2.CP3S & 1 ) << 4 )
86  | (( regs.r2.PowerDown & 1 ) << 5 )
87  | (( regs.r2.PhasePolarity & 1 ) << 6 )
88  | (( regs.r2.LDP & 1 ) << 7 )
89  | (( regs.r2.LDF & 1 ) << 8 )
90  | (( regs.r2.CPC & 0xf ) << 9 )
91  | (( regs.r2.DBufRFDiv & 1 ) << 13 )
92  | (( regs.r2.R & 0x3ff ) << 14 )
93  | (( regs.r2.REFinDIV2 & 1 ) << 24 )
94  | (( regs.r2.REFinMUL2 & 1 ) << 25 )
95  | (( regs.r2.MuxOut & 7 ) << 26 )
96  | (( regs.r2.NoiseSpurMode & 3 ) << 29 );
97  return(R );
98  case 3 :
99  R = 3
100  | (( regs.r3.ClkDiv & 0x0fff ) << 3 )
101  | (( regs.r3.ClkDivMode & 3 ) << 15 )
102  | (( regs.r3.CSR & 1 ) << 18 )
103  | (( regs.r3.CPC & 1 ) << 21 )
104  | (( regs.r3.ABPW & 1 ) << 22 )
105  | (( regs.r3.BandClkMode & 1 ) << 23 );
106  return(R );
107  case 4 :
108  R = 4
109  | (( regs.r4.RFOutPower & 3 ) << 3 )
110  | (( regs.r4.RFOutEnable & 1 ) << 5 )
111  | (( regs.r4.AuxOutPower & 3 ) << 6 )
112  | (( regs.r4.AuxOutEnable & 1 ) << 8 )
113  | (( regs.r4.AuxOutSel & 1 ) << 9 )
114  | (( regs.r4.MTLD & 1 ) << 10 )
115  | (( regs.r4.VCOPowerDown & 1 ) << 11 )
116  | (( regs.r4.BandClkDiv & 0xff ) << 12 )
117  | (( regs.r4.RFDivSel & 7 ) << 20 )
118  | (( regs.r4.FeedbackVCO & 1 ) << 23 );
119  return(R );
120  case 5 :
121  R = 5
122  | (( 3 ) << 19 ) // reserved high
123  | (( regs.r5.LDPinMode & 3 ) << 22 );
124  return(R);
125  }
126  return 0xffffffffU;
127 }

340 // PFD = REFin × [(1 + REfinMUL2)/(R × (1 + REFinDIV2))]
341 // REFin is the reference frequency input.
342 // REFinMUL2 is the REFin doubler bit (0 or 1).
343 // R is the RF reference division factor (1 to 1023).
344 // REFinDIV2 is the reference divide-by-2 bit (0 or 1).
345 MEMSPACE
346 double ADF4351_PFD(double REFin, int R)
347 {
348  double PFD = (uint32_t) REFin
349  * (regs.r2.REFinMUL2 ? 2 : 1)
350  / (R * (regs.r2.REFinDIV2 ? 2 : 1) );
351  return(PFD);
352 }

406 int ADF4351_Config(double RFout, double REFin, double ChannelSpacing, double *RFoutCalc )
407 {
408 
409  uint32_t div_gcd;
410  uint32_t r0_INT;
411  uint32_t r0_FRAC;
412  uint32_t r1_MOD;
413  uint32_t r1_Prescaler;
414  uint32_t r2_R;
415  uint32_t r4_RFDivSel;
416  uint32_t r4_BandClkDiv;
417  uint32_t RFoutDIV;
418  double N;
419  uint32_t N_min;
420 
421  double PFD;
422  double BandSelectClockFrequency;
423  double Fres;
424 
425  double dscale;
426  uint32 temp;
427 
428  *RFoutCalc = 0.0;
429 
447  // ==========================
448  if (RFout > ADF4351_RFOUT_MAX)
449  {
450 #if ADF4351_DEBUG & 2
451  printf("RFout > %.2f\n", (double) ADF4351_RFOUT_MAX);
452 #endif
453  return(ADF4351_RFout_RANGE);
454  }
455 
456  if (RFout < ADF4351_RFOUT_MIN)
457  {
458 #if ADF4351_DEBUG & 2
459  printf("RFout < %.2f\n", (double) ADF4351_RFOUT_MIN);
460 #endif
461  return(ADF4351_RFout_RANGE);
462  }
463 
464  if (REFin > ADF4351_REFIN_MAX)
465  {
466 #if ADF4351_DEBUG & 2
467  printf("REFin > %.2f\n", (double) ADF4351_REFIN_MAX);
468 #endif
469  return(ADF4351_REFin_RANGE);
470  }
471 
472  // ==========================
473  // Compute RFout divider and R4 register value
474  r4_RFDivSel = 0;
475  RFoutDIV = 1;
476  while(((RFout * RFoutDIV) < ADF4351_VCO_MIN) && RFoutDIV < 64)
477  {
478  RFoutDIV <<= 1;
479  r4_RFDivSel++;
480  }
481 
482  // Compute r1_prescale selector based on RFout
483  if(RFout > ADF4351_MAX_FREQ_45PRE)
484  {
485  r1_Prescaler = 1;
486  N_min = 75;
487  }
488  else
489  {
490  r1_Prescaler = 0;
491  N_min = 23;
492  }
493 
494 
495  // ==========================
496  // PFD = REFin × [(1 + REFinMUL2)/(R × (1 + REFinDIV2))]
497  // REFin is the reference frequency input.
498  // REFinMUL2 is the REFin doubler bit (0 or 1).
499  // r2_R is REFin division factor (1 to 1023).
500  // REFinDIV2 is the reference divide-by-2 bit (0 or 1).
501  r2_R = 1;
502  while(r2_R < 4096 )
503  {
504  PFD = ADF4351_PFD(REFin, r2_R);
505  if(PFD < ADF4351_PFD_MAX)
506  break;
507  r2_R++;
508  }
509 
510  if(r2_R == 4096)
511  {
512 #if ADF4351_DEBUG & 2
513  printf("r2_R == 4096\n");
514 #endif
515  return(ADF4351_R_RANGE);
516  }
517 
518  // ==========================
519  // Check for PFD range errors
520  if (regs.r3.BandClkMode == 0)
521  {
522  if( PFD > ADF4351_PFD_MAX )
523  {
524 #if ADF4351_DEBUG & 2
525  printf("PFD: %.2f > %u && BandClkMode == 0\n",
526  (double) PFD, (uint32_t) ADF4351_PFD_MAX);
527 #endif
528  return(ADF4351_PDF_RANGE);
529  }
530  }
531  else
532  {
533  if( PFD > ADF4351_PFD_MAX && r0_FRAC != 0)
534  {
535 #if ADF4351_DEBUG & 2
536  printf("PFD: %.2f > %u && BandClkMode == 0\n",
537  (double) PFD, (uint32_t) ADF4351_PFD_MAX);
538 #endif
539  return(ADF4351_PDF_RANGE);
540  }
541  if (PFD > 90 && r0_FRAC != 0)
542  {
543 #if ADF4351_DEBUG & 2
544  printf("PFD: %.2f > 90 Band Clock Mode && r0_FRAC != 0\n",
545  (double) PFD);
546 #endif
547  return(ADF4351_PDF_RANGE);
548  }
549  }
550 
551  // ==========================
552  // Compute N based on R4 feedback path select
553  if (regs.r4.FeedbackVCO)
554  N = ((double)RFout * (double)RFoutDIV) / (double) PFD;
555  else
556  N = ((double)RFout / (double) PFD);
557 
558  if(N < N_min || N > 65535U )
559  {
560 #if ADF4351_DEBUG & 2
561  printf("N %.2f out of range\n", (double) N);
562 #endif
563  return(ADF4351_N_RANGE);
564  }
565 
566  // ==========================
567  // Integer
568  r0_INT = (uint32_t) N;
569 
570  // r0_INt range check
571  if (r0_INT > 65535U)
572  {
573 #if ADF4351_DEBUG & 2
574  printf("INT: %u\n", (uint32_t) r0_INT);
575 #endif
576  return(ADF4351_INT_RANGE);
577  }
578 
579  // ==========================
580  // Modulus
581  // FIXME this solution, which works, disagrees with datasheet
582  // r1_MOD=REFin/Fres
583  // Fres = ChannelSpacing/RFoutDIV
584 
585  // Fres = VCO Channel Spacing
586  Fres = (double) RFoutDIV * ChannelSpacing;
587 
588  // FIXME Main_Form.cs disagrees with datasheet
589  // r1_MOD=(uint32_t) round((double) REFin / (double) Fres);
590  r1_MOD=(uint32_t) round((double) PFD / (double) ChannelSpacing );
591 
592  // ==========================
593  // Fractional
594  r0_FRAC = (uint32_t)round( ((double)N-(double)r0_INT)*(double) r1_MOD);
595 
596  // ==========================
597  // Reduce r1_MOD and r0_FRAC by greatest common divisor
598  div_gcd = ADF4351_GCD32(r1_MOD, r0_FRAC);
599  r1_MOD /= div_gcd;
600  r0_FRAC /= div_gcd;
601 
602  // FIXME - why is this needed ?
603  if (r1_MOD == 1)
604  r1_MOD = 2;
605 
606  // ==========================
607  // r1_MOD Range check
608  if(r1_MOD == 0 || r1_MOD > 4095U)
609  {
610 #if ADF4351_DEBUG & 2
611  printf("*MOD: %u, INT: %u, FRAC: %u\n", (uint32_t) r1_MOD, (uint32_t) r0_INT, (uint32_t) r0_FRAC);
612 #endif
613  return(ADF4351_MOD_RANGE);
614  }
615 
616  // ==========================
617  // r0_FRAC range check
618  if (r0_FRAC > 4095U)
619  {
620 #if ADF4351_DEBUG & 2
621  printf("MOD: %u, INT: %u, *FRAC: %u\n", (uint32_t) r1_MOD, (uint32_t) r0_INT, (uint32_t) r0_FRAC);
622 #endif
623  return(ADF4351_FRAC_RANGE);
624  }
625 
626  // ==========================
627  // Band Clock Divider
628  // FIXME Add user override to pick value
629  // FIXME perhaps we could consider setting regs.r3.BandClkMode based on PFD ?
630  if (regs.r3.BandClkMode == 0)
631  dscale = 8;
632  else
633  dscale = 2;
634  temp = (uint32_t)round(dscale * PFD);
635  if ((dscale * PFD - temp) > 0)
636  temp++;
637 
638  temp = (temp > 255) ? 255 : temp;
639  r4_BandClkDiv = temp;
640 
641  // ==========================
642  // Band Clock Frequency
643  // FYI temp will always be > 0
644  BandSelectClockFrequency = (PFD / (double)r4_BandClkDiv);
645 
646  // ==========================
647  // Band Clock Range Check
648  if (BandSelectClockFrequency > 500000.0)
649  {
650 #if ADF4351_DEBUG & 2
651  printf("Band Slect Clock Frequency %.2f > 500000\n",
652  (double) BandSelectClockFrequency);
653 #endif
654  return(ADF4351_BandSelectClockFrequency_RANGE);
655  }
656 
657  if ((BandSelectClockFrequency > 125000.0) & (regs.r3.BandClkMode))
658  {
659 #if ADF4351_DEBUG & 2
660  printf("Band Select Clock Frequency %.2f > 125000 && regs.r3.BandClkMode\n",
661  (double) BandSelectClockFrequency);
662 #endif
663  return(ADF4351_BandSelectClockFrequency_RANGE);
664  }
665 
666  // ==========================
667  // Noise Spur Mode
668  if ((regs.r2.NoiseSpurMode == ADF4351_LOW_SPUR_MODE) && (r1_MOD < 50))
669  {
670 #if ADF4351_DEBUG & 2
671  printf("regs.r2.NoiseSpurMode == ADF4351_LOW_SPUR_MODE) && (r1_MOD(%d) < 50\n",
672  (uint32_t) r1_MOD);
673 #endif
674  return(ADF4351_MOD_RANGE);
675  }
676 
677  // ==========================
678  // Write Registers
679  regs.r0.INT = r0_INT;
680  regs.r0.FRAC = r0_FRAC;
681  regs.r1.MOD = r1_MOD;
682  regs.r1.Prescaler = r1_Prescaler;
683 
684  regs.r2.R = r2_R;
685 
686  regs.r4.BandClkDiv = r4_BandClkDiv;
687  regs.r4.RFDivSel = r4_RFDivSel;
688 
689  // ==========================
690  // Compute actual RFout
