189  tft_reset_enable();

192  tft_delay_us(1000);
193 
194  /* Adafruit 9341 TFT Display Initialization */
195  tft_configRegister();
196 
197  /* Read the TFT ID value */
198  tft_ID = tft_readId();
199 
200  // fast SPI
201  tft_spi_init_fast();
202
