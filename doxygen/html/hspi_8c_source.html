99 void hspi_init(uint16_t prescale, int hwcs)
100 {
101 
102 // FIXME DEBUG
103 #if 1
104  // We only make changes if the speed actually changes
105  if(prescale == hspi_clock)
106  return;
107 #endif
108 
109  // If we have been called make sure all spi transactions have finished before
110  // we change anything
111 
112  if(_hspi_init_done)
113  hspi_waitReady();
114 
115  /* eagle_soc.h does not define MISO,MOSI,CLK and CS */
116  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTDI_U, 2); // HSPIQ MISO GPIO12
117  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTCK_U, 2); // HSPID MOSI GPIO13
118  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTMS_U, 2); // CLK GPIO14
119 
120  // HARDWARE SPI CS
121  if(hwcs)
122  PIN_FUNC_SELECT(PERIPHS_IO_MUX_MTDO_U, 2); // CS AUTOMATIC ONLY ON GPIO15
123 
124  if(prescale == 0)
125  {
126  WRITE_PERI_REG(PERIPHS_IO_MUX, 0x305); //set bit9
127  WRITE_PERI_REG(SPI_FLASH_CLOCK(HSPI), SPI_CLK_EQU_SYSCLK);
128  }
129  else
130  {
131  // prescale >= 1
132  WRITE_PERI_REG(PERIPHS_IO_MUX, 0x105); //clear bit9
133  WRITE_PERI_REG(SPI_FLASH_CLOCK(HSPI),
134  (((prescale - 1) & SPI_CLKDIV_PRE) << SPI_CLKDIV_PRE_S) |
135  ((1 & SPI_CLKCNT_N) << SPI_CLKCNT_N_S) |
136  ((0 & SPI_CLKCNT_H) << SPI_CLKCNT_H_S) |
137  ((1 & SPI_CLKCNT_L) << SPI_CLKCNT_L_S));
138  }
139 
140  WRITE_PERI_REG(SPI_FLASH_CTRL1(HSPI), 0);
141  hspi_clock = prescale;
142  _hspi_init_done = 1;
143 }

213 static void hspi_writeFIFO(uint8_t *write_data, int bytes)
214 {
215  uint8_t word_ind = 0;
216 
217  if(bytes > HSPI_FIFO_SIZE) // TODO set error status
218  return;
219 
220  hspi_setBits(bytes); // Update FIFO with number of bits we will send
221 
222 // First do a fast write with 32 bit chunks at a time
223  while(bytes >= 4)
224  {
225 // Cast both source and destination to 4 byte word pointers
226  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = \
227  ((uint32_t *)write_data) [word_ind];
228 
229  bytes -= 4;
230  word_ind++;
231  }
232 
233 // Next write remaining bytes (if any) to FIFO as a single word
234 // Note: We follow the good practice of avoiding access past the end of an array.
235 // (ie. protected memory, PIC using retw from flash ... could crash some CPU's)
236  if(bytes) // Valid counts are now: 3,2,1 bytes
237  {
238  uint32_t last_word = 0; // Last word to send
239  uint16_t byte_ind = word_ind << 2; // Convert to Byte index
240 
241 // Working MSB to LSB allows assigment to LSB without shifting
242 // Compiler can optimize powers of 8 into byte swaps, etc... (on some CPU's)
243  while(bytes--) // index is now 2,1,0 matching required storage offset
244  {
245  last_word <<= 8;
246 // 2,1,0
247  last_word |= ((uint32_t) 0xffU & write_data[byte_ind + bytes ]);
248  }
249 // Send last partial word
250  ((uint32_t *)SPI_FLASH_C0(HSPI)) [word_ind] = last_word;
251  }
252 }
