121 uint16_t XPT2046_read(uint8_t cmd)
125 {
126  uint8_t buf[3];
127  uint16_t val;
128 
129  // First byte to send is the command
130  buf[0] = cmd;
131  // 0 is ignored as a command NOOP, used here to read the reply from the ADC
132  buf[1] = 0;
133  buf[2] = 0;
134 
135  spi_begin(XPT2046_clock, XPT2046_CS);
136  // Send three bytes and read the result
137  spi_TXRX_buffer(buf,3);
138  spi_end(XPT2046_CS);
139 
140  // buf[0] - Ignore data read back from the first command byte - is has no valid data
141 
142  // Extract the ADC 12 bit reply - ADC result starts one bit AFTER the MSB bit position
143  val = buf[1]; // MSB
144  val <<= 8;
145  val |= buf[2]; // LSB
146 
147  // Align 12 bit result to LSB bit position
148  val >>= 3;
149 
150  if(val < 0)
151  val = 0;
152  if(val >4095)
153  val = 4095;
154  return(val);
155 }

407 void XPT2046_task()
408 {
409  uint16_t X,Y;
410  int T;
411 
412  T = XPT2046_xy_filtered((uint16_t *)&X, (uint16_t *)&Y);
413 
414  // Key debounce state machine
415  switch(xpt2046.state)
416  {
417  // Full init
418  case 0:
419  xpt2046.state = 1;
420  xpt2046.ms = 0;
421  xpt2046.ind = 0;
422  xpt2046.head = 0;
423  xpt2046.tail = 0;
424  // Fall through to state 1
425  case 1:
426  // FIXME this can be a lower count for a button as we are not drawing here
427  if(T) // key is down and noise level is in range
428  {
429  // wait for key down debounce time before taking a sample
430  if(++xpt2046.ms < XPT2046_DEBOUNCE)
431  break;
432 
433  if(xpt2046.ind < XPT2046_EVENTS )
434  {
435  xpt2046.XQ[xpt2046.head] = X;
436  xpt2046.YQ[xpt2046.head] = Y;
437  if(++xpt2046.head >= XPT2046_EVENTS)
438  xpt2046.head = 0;
439  xpt2046.ind++;
440  xpt2046.state = 2;
441  xpt2046.ms = 0;
442  }
443  else
444  {
445  xpt2046.ms = 0;
446  }
447  } // if (T)
448  else // Restart timer and wait for touch debounce time
449  {
450  xpt2046.ms = 0;
451  }
452  break;
453 
454  // Wait for touch release debounce time
455  case 2:
456  // Released ?
457  if(T == 0)
458  {
459  // Debounce release time - valid key depress cycle done
460  if(++xpt2046.ms >= XPT2046_DEBOUNCE)
461  {
462  xpt2046.ms = 0;
463  xpt2046.state = 1;
464  }
465  }
466  else // Still depressed
467  {
468  xpt2046.ms = 0;
469  }
470  break;
471 
472 
473  // Error INIT
474  default:
475  xpt2046.state = 0;
476  break;
477  }
478 }

518 int sdev(uint16_t *samples, int size, sdev_t *Z)
519 {
520  int i;
521  float val, delta, sum, sqsum;
522 
523  Z->min = ~0;
524  Z->max = 0;
525 
526 // error in sample size ?
527  if(size < 2)
528  return(0);
529 
530  sum = 0.0;
531  for(i=0; i<size ; i++) {
532  val = samples[i];
533  if(Z->min > val)
534  Z->min = val;
535  if(Z->max < val)
536  Z->max = val;
537  sum += val;
538  }
539  Z->mean = sum / (float) size;
540  sqsum = 0.0;
541  for(i=0;i<size;++i) {
542  val = samples[i];
543  if(val < 0 || val > 4095)
544  continue; // reject
545  delta = val - Z->mean;
546  sqsum += (delta * delta);
547  }
548  val = sqsum / (float) (i - 1);
549  Z->sdev = sqrt(val);
550  return(1);
551 }
